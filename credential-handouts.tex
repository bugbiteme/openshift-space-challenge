\documentclass{article}
\pagestyle{empty}
\usepackage[a4paper, margin=1in]{geometry}
\usepackage{csvsimple}
\usepackage{tikz}
\usepackage{pgffor}

% Define a box for each slip
\newcommand{\printslip}[2]{%
    \noindent
    \begin{tikzpicture}
        \draw[rounded corners=5mm] (0,0) rectangle (\linewidth,-.5\textheight);
        \node at (.5\linewidth,-.25\textheight) {
            \begin{minipage}{.8\linewidth}
                \centering
                \large
                \textbf{Username:} #1 \\
                \textbf{Password:} #2
            \end{minipage}
        };
    \end{tikzpicture}

    % Add space after the first slip but not the second
    \ifnum\thecsvrow=1
        \vspace{\fill}
    \else
        \newpage
    \fi
}

\begin{document}

\csvreader[head to column names]{credentials.csv}{}{%
    \printslip{\username}{\password}
}

\end{document}
