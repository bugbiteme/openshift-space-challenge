% No page numbers
\pagestyle{empty}

\newcommand{\printslip}[2]{%
    \noindent
    \begin{tikzpicture}
        % Draw a rounded rectangle border
        \draw[rounded corners=10pt, line width=1pt] (0,0) rectangle (\linewidth,-.5\textheight);

        % Diagonal watermark
        \node[gray, scale=5, rotate=45] at (0.5\linewidth, -.3\textheight) {SECRET};

        % Horizontal separator line
        \draw[dashed, gray] (0.1\linewidth, -.25\textheight) -- (0.9\linewidth, -.25\textheight);

        % Print Username & Password
        \node at (.6\linewidth,-.18\textheight) {
            \begin{minipage}{.4\linewidth}
                \large \textbf{Username:} \\
                #1 \\
                \large \textbf{Password:} \\
                #2
            \end{minipage}
        };
    \end{tikzpicture}

    % Add space after the first slip but not the second
    \ifnum\thecsvrow=1
        \vspace{\fill}
    \else
        \newpage
    \fi
}

\begin{document}

\csvreader[head to column names]{credentials.csv}{}{%
    \printslip{\username}{\password}
}

\end{document}
